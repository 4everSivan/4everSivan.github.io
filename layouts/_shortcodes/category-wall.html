{{/* 分类导航列表: 自动遍历 /docs 全部一级分类, 按文档计数降序展示; 无已发布文档的分类不展示. @example {{< category-wall >}} */ -}} {{- $docs := site.GetPage "/docs" -}} {{- if not $docs -}} {{- errorf "category-wall: controlled /docs content root is missing" -}} {{- end -}} {{- /* 收集一级分类: 标题, 链接, 递归文档数与最近一篇更新文档 */ -}} {{- $items := slice -}} {{- range $docs.Sections -}} {{- $pages := .RegularPagesRecursive -}} {{- if $pages -}} {{- $items = $items | append (dict "title" .Title "link" .RelPermalink "count" (len $pages) "latest" (first 1 (sort $pages "Lastmod" "desc")) ) -}} {{- end -}} {{- end -}}
{{- range $item := sort $items "count" "desc" -}} {{ $item.title }} {{ $item.count }} 篇 {{- with index $item.latest 0 -}} {{ .Title }} {{- end -}} {{- end -}}
